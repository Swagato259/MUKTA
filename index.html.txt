    <script>
        // Create floating hearts
        function createHearts() {
            const container = document.getElementById('hearts-container');
            const heartCount = 30;
            
            for (let i = 0; i < heartCount; i++) {
                const heart = document.createElement('div');
                heart.classList.add('heart');
                heart.innerHTML = '❤️';
                heart.style.left = Math.random() * 100 + 'vw';
                heart.style.fontSize = (Math.random() * 20 + 10) + 'px';
                heart.style.animationDuration = (Math.random() * 15 + 10) + 's';
                heart.style.animationDelay = Math.random() * 5 + 's';
                container.appendChild(heart);
            }
        }
        
        // Navigation functions
        function showPage(pageId) {
            document.querySelectorAll('.page').forEach(page => {
                page.style.display = 'none';
            });
            document.getElementById(pageId).style.display = 'block';
        }
        
        function nextPage(currentPage) {
            const pages = ['page1', 'page2', 'page3', 'page4'];
            const currentIndex = pages.indexOf(currentPage);
            if (currentIndex < pages.length - 1) {
                showPage(pages[currentIndex + 1]);
            }
        }
        
        function prevPage(currentPage) {
            const pages = ['page1', 'page2', 'page3', 'page4'];
            const currentIndex = pages.indexOf(currentPage);
            if (currentIndex > 0) {
                showPage(pages[currentIndex - 1]);
            }
        }
        
        // Auto-scroll function
        function autoScroll() {
            const container = document.querySelector('.message-container');
            let scrollPos = 0;
            const scrollInterval = setInterval(() => {
                scrollPos += 1;
                container.scrollTop = scrollPos;
                
                if (scrollPos >= container.scrollHeight - container.clientHeight) {
                    clearInterval(scrollInterval);
                }
            }, 50);
        }
        
        // Event Listeners
        document.addEventListener('DOMContentLoaded', function() {
            createHearts();
            
            // Next button for page1
            document.getElementById('next-btn1').addEventListener('click', function() {
                nextPage('page1');
            });
            
            // Door handle click
            document.getElementById('door-handle').addEventListener('click', function() {
                document.getElementById('knock-btn').style.display = 'block';
                this.style.transform = 'rotate(30deg)';
                this.style.boxShadow = '0 0 20px rgba(255, 213, 79, 1)';
            });
            
            // Knock button
            document.getElementById('knock-btn').addEventListener('click', function() {
                nextPage('page2');
                
                // Animate messages
                setTimeout(() => {
                    const messages = document.querySelectorAll('.message');
                    messages.forEach((msg, index) => {
                        setTimeout(() => {
                            msg.classList.add('visible');
                        }, index * 600);
                    });
                    
                    // Auto-scroll after messages appear
                    setTimeout(autoScroll, messages.length * 600 + 1000);
                }, 500);
            });
            
            // Promise button
            document.getElementById('promise-btn').addEventListener('click', function() {
                nextPage('page3');
            });
            
            // Navigation buttons
            document.getElementById('prev-btn').addEventListener('click', function() {
                const currentPage = document.querySelector('.page[style*="block"]').id;
                prevPage(currentPage);
            });
            
            document.getElementById('restart-btn').addEventListener('click', function() {
                showPage('page1');
            });
        });
    </script>
